Archivo deployment-ms-eventos.yaml:

apiVersion: apps/v1
kind: Deployment
metadata:
  name: ms-eventos
spec:
  replicas: 1
  selector:
    matchLabels:
      app: ms-eventos
  template:
    metadata:
      labels:
        app: ms-eventos
    spec:
      containers:
      - name: ms-eventos
        image: jonathan0284/trabajo-final/ms-eventos:1.3
        ports:
        - containerPort: 9090
        env:
        - name: DB_USERNAME
          valueFrom:
            secretKeyRef:
              name: db-credentials
              key: DB_USERNAME
        - name: DB_PASSWORD
          valueFrom:
            secretKeyRef:
              name: db-credentials
              key: DB_PASSWORD
        - name: APP_NAME
          valueFrom:
            configMapKeyRef:
              name: ms-eventos-config
              key: APP_NAME
        - name: SERVER_PORT
          valueFrom:
            configMapKeyRef:
              name: ms-eventos-config
              key: SERVER_PORT
        - name: DB_URL
          valueFrom:
            configMapKeyRef:
              name: ms-eventos-config
              key: DB_URL
        - name: DB_DRIVER
          valueFrom:
            configMapKeyRef:
              name: ms-eventos-config
              key: DB_DRIVER
        - name: JPA_DDL_AUTO
